inconsistent_exit_date_long = dbGetQuery(con, "SELECT distinct a.unique_id
                                                FROM student_mobility_fields_2005_2015 a

unique_id_list = paste0("'", paste0(inconsistent_exit_date_long$unique_id, collapse = "','"), "'")
